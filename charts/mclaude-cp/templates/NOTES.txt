Before the platform is functional, create the required Secrets:

  # Postgres password
  kubectl create secret generic mclaude-postgres \
    --namespace {{ include "mclaude-cp.namespace" . }} \
    --from-literal=postgres-password="$(openssl rand -base64 32)"

  # Control-plane credentials (replace <POSTGRES_PASSWORD> with the password above)
  kubectl create secret generic mclaude-control-plane \
    --namespace {{ include "mclaude-cp.namespace" . }} \
    --from-literal=admin-token="$(openssl rand -base64 32)" \
    --from-literal=database-url="<POSTGRES_CONNECTION_STRING>"
